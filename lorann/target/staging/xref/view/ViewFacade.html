36  	 * Initialize the map.
37  	 * displayMessage create the windows.
38  	 * Recovers the database.
39  	 */
40  	@Override
41  	public final void displayMessage(final String message) {
42  		
43  		System.out.println("- Affichage Map");
44  
45  		StringBuffer map = new StringBuffer(message);
46  		map.delete(0, 4);
47  
48  		/** The JFrame window.*/
49  		JFrame window = new JFrame();
50  		/** The level */
51  		String level = new String(map);
52  
53  		/** Create the window. */
54  		window.setSize(769, 468);
55  		window.setTitle("Lorann Game");
56  		window.setLocationRelativeTo(null);
57  		window.setResizable(false);
58  		window.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
59  		
60  		/** Show the map. */
61  		JPanel pan = new JPanel();
62  		pan.setBackground(Color.BLACK);
63  		pan.setLayout(new GridLayout(12, 20));
64  
65  		/** Change the letters on the map to sprite. */
66  		for (int i = 0; i != 240; i++) {
67  			char caracter = level.charAt(i);
68  			switch (caracter) {
69  			case 'A':
70  				Icon laurann = new ImageIcon("../sprite/Lorann.gif");

85  				c.setIcon(crystall_ball);
86  				pan.add(c);
87  				break;

90  				JLabel gc = new JLabel();
